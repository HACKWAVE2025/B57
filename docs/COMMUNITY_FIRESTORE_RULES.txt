// Firestore Security Rules for Community Feature
// Add these rules to your existing firestore.rules file

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Community Posts Collection
    match /community_posts/{postId} {
      // Anyone authenticated can read posts
      allow read: if request.auth != null;
      
      // Only authenticated users can create posts
      allow create: if request.auth != null 
        && request.resource.data.authorId == request.auth.uid
        && request.resource.data.likes >= 0
        && request.resource.data.comments >= 0
        && request.resource.data.shares >= 0
        && request.resource.data.views >= 0;
      
      // Only post author can update or delete their own posts
      allow update: if request.auth != null 
        && (resource.data.authorId == request.auth.uid
            || request.resource.data.keys().hasOnly(['likes', 'likedBy', 'views', 'viewedBy', 'shares', 'comments', 'updatedAt']));
      
      allow delete: if request.auth != null 
        && resource.data.authorId == request.auth.uid;
    }
    
    // Community Events Collection
    match /community_events/{eventId} {
      // Anyone authenticated can read events
      allow read: if request.auth != null;
      
      // Only authenticated users can create events
      allow create: if request.auth != null 
        && request.resource.data.createdBy == request.auth.uid;
      
      // Event creator can update or anyone can join/leave (update attendees)
      allow update: if request.auth != null 
        && (resource.data.createdBy == request.auth.uid
            || request.resource.data.keys().hasOnly(['attendees']));
      
      // Only event creator can delete
      allow delete: if request.auth != null 
        && resource.data.createdBy == request.auth.uid;
    }
    
    // Community Leaderboard Collection
    match /community_leaderboard/{userId} {
      // Anyone authenticated can read leaderboard
      allow read: if request.auth != null;
      
      // Users can create their own leaderboard entry
      allow create: if request.auth != null 
        && userId == request.auth.uid
        && request.resource.data.userId == request.auth.uid;
      
      // Only the user or system can update their own entry
      allow update: if request.auth != null 
        && userId == request.auth.uid;
      
      // Users cannot delete leaderboard entries
      allow delete: if false;
    }
    
    // Community Resources Collection
    match /community_resources/{resourceId} {
      // Anyone authenticated can read resources
      allow read: if request.auth != null;
      
      // Only authenticated users can create resources
      allow create: if request.auth != null 
        && request.resource.data.authorId == request.auth.uid;
      
      // Only resource author can update or anyone can download (update downloads)
      allow update: if request.auth != null 
        && (resource.data.authorId == request.auth.uid
            || request.resource.data.keys().hasOnly(['downloads', 'downloadedBy']));
      
      // Only resource author can delete
      allow delete: if request.auth != null 
        && resource.data.authorId == request.auth.uid;
    }
    
    // Community Comments Collection
    match /community_comments/{commentId} {
      // Anyone authenticated can read comments
      allow read: if request.auth != null;
      
      // Only authenticated users can create comments
      allow create: if request.auth != null 
        && request.resource.data.authorId == request.auth.uid;
      
      // Only comment author can update their own comments
      allow update: if request.auth != null 
        && resource.data.authorId == request.auth.uid;
      
      // Only comment author can delete their own comments
      allow delete: if request.auth != null 
        && resource.data.authorId == request.auth.uid;
    }
  }
}
